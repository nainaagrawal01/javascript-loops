# javascript-loops
for loop-------

for(let i=1;i<5;i++){
   console.log(i);
 }


while loop--------

let j=1;
while(j<5){
    console.log(j);
    j++;
}


do while loop-------------

let k=1;
do{
    console.log(k);
    k++
}
while(k<5);


for in loop-----------

const num = [1, 2, 3, 4, 5];
for (const index in num) {
    console.log(`${index}: ${num[index]}`);
}
